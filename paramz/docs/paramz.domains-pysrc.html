31  """ 
32  (Hyper-)Parameter domains defined for :py:mod:`paramz.transformations`. 
33  These domains specify the legitimate realm of the parameters to live in. 
34   
35  :const:`~paramz.domains._REAL` : 
36      real domain, all values in the real numbers are allowed 
37   
38  :const:`~paramz.domains._POSITIVE`: 
39      positive domain, only positive real values are allowed 
40   
41  :const:`~paramz.domains._NEGATIVE`: 
42      same as :const:`~paramz.domains._POSITIVE`, but only negative values are allowed 
43   
44  :const:`~paramz.domains._BOUNDED`: 
45      only values within the bounded range are allowed, 
46      the bounds are specified withing the object with the bounded range 
47  """ 
48   
49  _REAL = 'real' 
50  _POSITIVE = "positive" 
51  _NEGATIVE = 'negative' 
52  _BOUNDED = 'bounded'
